1. is() -> clone
	$('#clone').on('click', function(){
		$('#d1 p').on('click', function(){
			if($(this).parent().is('.dd')){
				$(this).clone().appendTo($(this).parent()).css('color', 'red');
			}
		})
	})



2. is() -> clone(true)
	$('#cloneT').on('click', function(){
		$('#d1 p').on('click', function(){
			if($(this).parent().is('.dd')){
				$(this).clone(true).appendTo($(this).parent()).css('color', 'red');
			}
		})
	})
