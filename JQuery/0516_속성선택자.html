$(function(){
	//버튼이 한 개일 때
	//$('#btn1'), $('input#btn1'), $('input[type=button]'), $('input[value=확인]'), $('input[id=btn1]') 다 가능
	$('#btn1').on('click', function(){
		//title 속성을 가진 p요소의 배경색 변경
		$('p[title]').css('background', 'lime');
		
        //type 속성이 password 인 요소의 테두리 변경
		$('input[type=password]').css('border', '5px solid pink');
	})
})

$('#btn2').on('click', function(){
		
		//type이 button인 input요소의 배경색 변경 
		$('#d2 input[type=button]').css('background', 'pink');
		
		//href가 mailto 로 시작하는 a 요소의 배경색 변경  
		$('a[href ^=mailto]').css('background', 'green');
		
		//id 가 img1인것만 제외한(아닌) img요소의 테두리 변경 
		$('img[id!=img1]').css('border', '5px solid grey');
		
		//src가 png로 끝나는 img 요소의 테두리 변경 
		$('img[src $= png]').css('border', '10px solid black');
		
		//type이 text인 input 요소의 테두리 
		$('input[type=text]').css('border', '5px double red');
		
		//title 이 named 단어(공백구분)를 포함한 img 요소 의 테두리  
		$('img[title~=named]').css('border', '10px dotted blue');
		
		//href가 power2c를 포함하는 모든 a 요소  의 밑줄 없음  
		$('a[href*=power2c]').css('text-decoration', 'none');
		
		//id가 little 이거나 little- 로 시작하는  img요소 의 테두리  
		$('img[id|=little]').css('border', '7px solid yellow');
		
		//alt 속성을 갖는 이미지 요소 의 테두리- 2px 단선  레드 
		$('img[alt]').css('border', '2px solid red');
		
})
