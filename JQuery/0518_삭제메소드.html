empty() : 선택 대상의 자식 요소들만 삭제
remove() : 선택 대상과 그 자식 요소 모두 다 삭제

-------------------------------------------------------------------------

	btn = $(this).attr('value');
	if(btn == "empty"){
		$('#d1').empty();
	}else if(btn == "remove"){
		$('#d1').remove();
	}
